1. save()함수에서 target = nullptr 이 필요한지 확인 필요
2. 복사 생성자는 PersonInfo 에서만 필요한지 확인 필요
3. 복사 생성자를 추가하면 디폴트 생성자도 추가해야 하는지 확인 필요
4. PersonalInfo의 멤버 변수를 private 로하면 외부에서 직접 접근이 되지 않는다.

class PersonalInfo {
public:
	string name;
	string phone;
	string address;
public:
	// default constructor
	PersonalInfo();
	PersonalInfo(string, string, string);

	// copy constructor
	PersonalInfo(PersonalInfo &info);

	// overloading operators
	PersonalInfo& operator= (PersonalInfo& info);
	bool operator== (PersonalInfo& info);
	ostream& operator<< (ostream& os);
};

PersonalInfo::PersonalInfo() {}

PersonalInfo::PersonalInfo(string name, string phone, string address) {
	this->name = name;
	this->phone = phone;
	this->address = address;
}
